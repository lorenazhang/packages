WITH call_data AS
(
    SELECT
        LOWER(TRIM(connid)) AS connid,
        interactionid,
        primary_lob,
        sublob,
        calldriver1,
        CAST(interactiondatetime AS DATE) AS interaction_dt,

        /* YYYYMM */
        CAST(EXTRACT(YEAR FROM CAST(interactiondatetime AS DATE)) AS CHAR(4))
        || 
        CASE 
            WHEN EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) < 10 
            THEN '0' || CAST(EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) AS CHAR(1))
            ELSE CAST(EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) AS CHAR(2))
        END AS month_id,

        /* YYYYMMDD */
        CAST(EXTRACT(YEAR FROM CAST(interactiondatetime AS DATE)) AS CHAR(4))
        ||
        CASE 
            WHEN EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) < 10 
            THEN '0' || CAST(EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) AS CHAR(1))
            ELSE CAST(EXTRACT(MONTH FROM CAST(interactiondatetime AS DATE)) AS CHAR(2))
        END
        ||
        CASE 
            WHEN EXTRACT(DAY FROM CAST(interactiondatetime AS DATE)) < 10 
            THEN '0' || CAST(EXTRACT(DAY FROM CAST(interactiondatetime AS DATE)) AS CHAR(1))
            ELSE CAST(EXTRACT(DAY FROM CAST(interactiondatetime AS DATE)) AS CHAR(2))
        END AS day_id,

        /* Monday week start */
        CAST(interactiondatetime AS DATE)
          - ((CAST(interactiondatetime AS DATE) - DATE '1900-01-01') MOD 7) AS week_start

    FROM EIW_U_CEDA_VOCAL_V1.AMP_CALLDRIVERS2_0
    WHERE model_lang = 'English'
      AND calldriver1 NOT IN ('No Conversation', 'No Reason Disclosed')
      AND connid IS NOT NULL

      /* past 13 months */
      AND CAST(interactiondatetime AS DATE) >= ADD_MONTHS(CURRENT_DATE, -13)
      AND CAST(interactiondatetime AS DATE) < CURRENT_DATE
),

comp_data AS
(
    SELECT
        LOWER(TRIM(conn_id)) AS connid,
        case_nr,
        complaint_id,
        complaint_type_desc,
        CAST(wf_recd_dt AS DATE) AS wf_recd_dt,
        CAST(created_dt AS DATE) AS created_dt

    FROM EIW_I_CEDA_S1.COMPLAINTS_ECMP_REGISTRY_DLY_CS
    WHERE conn_id IS NOT NULL

      /* complaint received date range */
      AND CAST(wf_recd_dt AS DATE) >= ADD_MONTHS(CURRENT_DATE, -13)
      AND CAST(wf_recd_dt AS DATE) < CURRENT_DATE
)

SELECT
    c.week_start,
    c.month_id,
    c.calldriver1,
    c.primary_lob,
    c.sublob,

    COUNT(*) AS call_count,

    SUM(CASE 
            WHEN p.case_nr IS NOT NULL THEN 1 
            ELSE 0 
        END) AS complaint_count,

    SUM(CASE 
            WHEN p.complaint_type_desc = 'Service Complaint' THEN 1 
            ELSE 0 
        END) AS service_complaint_count,

    SUM(CASE 
            WHEN p.complaint_type_desc IN 
                 ('LOB Escalated Complaint', 'EO Escalated Complaint') THEN 1 
            ELSE 0 
        END) AS escalated_complaint_count,

    /* conversion rates */
    CAST(SUM(CASE 
                WHEN p.case_nr IS NOT NULL THEN 1 
                ELSE 0 
             END) AS DECIMAL(18,6)) 
        / NULLIFZERO(COUNT(*)) AS complaint_conversion_rate,

    CAST(SUM(CASE 
                WHEN p.complaint_type_desc = 'Service Complaint' THEN 1 
                ELSE 0 
             END) AS DECIMAL(18,6)) 
        / NULLIFZERO(COUNT(*)) AS service_conversion_rate,

    CAST(SUM(CASE 
                WHEN p.complaint_type_desc IN 
                     ('LOB Escalated Complaint', 'EO Escalated Complaint') THEN 1 
                ELSE 0 
             END) AS DECIMAL(18,6)) 
        / NULLIFZERO(COUNT(*)) AS escalated_conversion_rate,

    AVG(
        CASE
            WHEN p.created_dt IS NOT NULL 
             AND p.wf_recd_dt IS NOT NULL
            THEN p.created_dt - p.wf_recd_dt
            ELSE NULL
        END
    ) AS avg_gap_days

FROM call_data c
LEFT JOIN comp_data p
    ON c.connid = p.connid

GROUP BY
    c.week_start,
    c.month_id,
    c.calldriver1,
    c.primary_lob,
    c.sublob

ORDER BY
    c.calldriver1,
    c.primary_lob,
    c.sublob,
    c.week_start;
