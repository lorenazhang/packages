WITH call_data AS
(
    SELECT
        LOWER(TRIM(connid)) AS connid,
        interactionid,
        primary_lob,
        sublob,
        calldriver1,
        interactiondatetime,

        /* YYYYMM */
        SUBSTR(interactiondatetime, 1, 4)
        || SUBSTR(interactiondatetime, 6, 2) AS month_id,

        /* YYYYMMDD */
        SUBSTR(interactiondatetime, 1, 4)
        || SUBSTR(interactiondatetime, 6, 2)
        || SUBSTR(interactiondatetime, 9, 2) AS day_id,

        /* Monday week start */
        CAST(SUBSTR(interactiondatetime, 1, 10) AS DATE FORMAT 'YYYY-MM-DD')
          - ((CAST(SUBSTR(interactiondatetime, 1, 10) AS DATE FORMAT 'YYYY-MM-DD') - DATE '1900-01-01') MOD 7) AS week_start

    FROM EIW_U_CEDA_VOCAL_V1.AMP_CALLDRIVERS2_0
    WHERE model_lang = 'English'
      AND calldriver1 NOT IN ('No Conversation', 'No Reason Disclosed')
      AND connid IS NOT NULL
      AND interactiondatetime >= '2026-03-15'
      AND interactiondatetime <  '2026-06-01'
),

comp_data AS
(
    SELECT
        LOWER(TRIM(conn_id)) AS connid,
        case_nr,
        complaint_id,
        complaint_type_desc,
        CAST(wf_recd_dt AS DATE) AS wf_recd_dt,
        CAST(created_dt AS DATE) AS created_dt

    FROM EIW_I_CEDA_S1.COMPLAINTS_ECMP_REGISTRY_DLY_CS
    WHERE conn_id IS NOT NULL
      AND wf_recd_dt >= DATE '2026-03-15'
      AND wf_recd_dt <  DATE '2026-06-01'
)

SELECT
    c.week_start,
    c.month_id,
    c.calldriver1,
    c.primary_lob,
    c.sublob,

    COUNT(*) AS call_count,

    SUM(CASE WHEN p.case_nr IS NOT NULL THEN 1 ELSE 0 END) AS complaint_count,

    SUM(CASE 
            WHEN p.complaint_type_desc = 'Service Complaint' THEN 1 
            ELSE 0 
        END) AS service_complaint_count,

    SUM(CASE 
            WHEN p.complaint_type_desc IN ('LOB Escalated Complaint', 'EO Escalated Complaint') THEN 1 
            ELSE 0 
        END) AS escalated_complaint_count,

    CAST(SUM(CASE WHEN p.case_nr IS NOT NULL THEN 1 ELSE 0 END) AS DECIMAL(18,6))
        / NULLIFZERO(COUNT(*)) AS complaint_conversion_rate,

    CAST(SUM(CASE WHEN p.complaint_type_desc = 'Service Complaint' THEN 1 ELSE 0 END) AS DECIMAL(18,6))
        / NULLIFZERO(COUNT(*)) AS service_conversion_rate,

    CAST(SUM(CASE 
                WHEN p.complaint_type_desc IN ('LOB Escalated Complaint', 'EO Escalated Complaint') THEN 1 
                ELSE 0 
             END) AS DECIMAL(18,6))
        / NULLIFZERO(COUNT(*)) AS escalated_conversion_rate,

    AVG(
        CASE
            WHEN p.created_dt IS NOT NULL 
             AND p.wf_recd_dt IS NOT NULL
            THEN p.created_dt - p.wf_recd_dt
            ELSE NULL
        END
    ) AS avg_gap_days

FROM call_data c
LEFT JOIN comp_data p
    ON c.connid = p.connid

GROUP BY
    c.week_start,
    c.month_id,
    c.calldriver1,
    c.primary_lob,
    c.sublob

ORDER BY
    c.calldriver1,
    c.primary_lob,
    c.sublob,
    c.week_start;
