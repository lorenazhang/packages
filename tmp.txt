WITH call_data AS (
    SELECT
        LOWER(TRIM(connid)) AS connid,
        interactionid,
        primary_lob,
        sublob,
        calldriver1,
        interactiondatetime,

        /* month */
        TO_CHAR(CAST(interactiondatetime AS DATE), 'YYYYMM') AS month,

        /* Monday week start */
        CAST(interactiondatetime AS DATE)
            - ((CAST(interactiondatetime AS DATE) - DATE '1900-01-01') MOD 7) AS week,

        /* day */
        TO_CHAR(CAST(interactiondatetime AS DATE), 'YYYYMMDD') AS day

    FROM EIW_U_CEDA_VOCAL_V1.AMP_CALLDRIVERS2_0
    WHERE model_lang = 'English'
      AND calldriver1 NOT IN ('No Conversation', 'No Reason Disclosed')
      AND connid IS NOT NULL
      AND interactiondatetime >= ADD_MONTHS(CURRENT_DATE, -13)
      AND interactiondatetime < CURRENT_DATE
),

comp_data AS (
    SELECT
        LOWER(TRIM(conn_id)) AS connid,
        case_nr,
        complaint_id,
        complaint_type_desc,
        wf_recd_dt,
        created_dt,

        CASE
            WHEN case_nr IS NOT NULL THEN 1
            ELSE 0
        END AS comp_flag,

        CASE
            WHEN complaint_type_desc = 'Service Complaint' THEN 1
            ELSE 0
        END AS service_flag,

        CASE
            WHEN complaint_type_desc IN (
                'LOB Escalated Complaint',
                'EO Escalated Complaint'
            ) THEN 1
            ELSE 0
        END AS escalated_flag

    FROM EIW_I_CEDA_S1.COMPLAINTS_ECMP_REGISTRY_DLY_CS
    WHERE conn_id IS NOT NULL
      AND wf_recd_dt >= ADD_MONTHS(CURRENT_DATE, -13)
      AND wf_recd_dt < CURRENT_DATE
),

merged AS (
    SELECT
        c.week,
        c.month,
        c.day,
        c.calldriver1,
        c.primary_lob,
        c.sublob,
        c.connid,
        c.interactionid,

        COALESCE(p.comp_flag, 0) AS comp_flag,
        COALESCE(p.service_flag, 0) AS service_flag,
        COALESCE(p.escalated_flag, 0) AS escalated_flag,

        CASE
            WHEN p.created_dt IS NOT NULL
             AND p.wf_recd_dt IS NOT NULL
            THEN CAST(p.created_dt AS DATE) - CAST(p.wf_recd_dt AS DATE)
            ELSE NULL
        END AS gap_days

    FROM call_data c
    LEFT JOIN comp_data p
      ON c.connid = p.connid
)

SELECT
    week,
    calldriver1,
    primary_lob,
    sublob,

    COUNT(*) AS call_count,

    SUM(comp_flag) AS complaint_count,
    SUM(service_flag) AS service_complaint_count,
    SUM(escalated_flag) AS escalated_complaint_count,

    AVG(comp_flag) AS complaint_conversion_rate,
    AVG(service_flag) AS service_conversion_rate,
    AVG(escalated_flag) AS escalated_conversion_rate,

    MEDIAN(gap_days) AS median_gap_days

FROM merged
GROUP BY
    week,
    calldriver1,
    primary_lob,
    sublob

ORDER BY
    calldriver1,
    primary_lob,
    sublob,
    week;
