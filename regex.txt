CREATE VOLATILE TABLE vt_interaction_id AS
(
    SELECT TOP 5
        interaction_id
    FROM SBX_CEDA_POC.wf_lob_byodb_full
) WITH DATA
PRIMARY INDEX (interaction_id)
ON COMMIT PRESERVE ROWS;


SELECT
    b.load_date,
    b.interaction_id,
    b.interaction_dt,
    b.transcript_full,
    m.primary_lob,
    m.groupname,
    m.sublob
FROM SBX_CEDA_POC.wf_lob_byodb_full b
JOIN vt_interaction_id v
    ON b.interaction_id = v.interaction_id
LEFT JOIN SBX_CEDA_POC.wf_lob_metadata_vw m
    ON b.interaction_id = m.interaction_id;
