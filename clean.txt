compiled_mapping = []

for category, product, kws in mapping:
    pattern = re.compile(
        r'\b(' + '|'.join(re.escape(k) for k in kws) + r')\b',
        flags=re.IGNORECASE
    )
    compiled_mapping.append((category, product, pattern))
