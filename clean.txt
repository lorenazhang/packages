import json
import pandas as pd
import time

BATCH_SIZE = 200


def clean_llm_json(response_text):
    """
    Remove markdown fences and parse JSON.
    """
    text = response_text.strip()
    text = text.replace("```json", "").replace("```", "").strip()

    # in case model adds text before JSON
    start_obj = text.find("{")
    start_arr = text.find("[")

    if start_obj == -1 and start_arr == -1:
        raise ValueError("No JSON object or array found in response")

    if start_obj == -1:
        start = start_arr
    elif start_arr == -1:
        start = start_obj
    else:
        start = min(start_obj, start_arr)

    text = text[start:]

    return json.loads(text)


def build_batch_prompt(batch_df):
    summary_notes_concat = "\n\n".join(
        f"[COMPLAINT {i+1}]\n{note}"
        for i, note in enumerate(batch_df["summary_note"].fillna(""))
    )

    prompt = f"""
Task:
Read the complaint summary notes and identify the main reasons why customers called.

Instructions:
- Each block labeled [COMPLAINT n] is one complaint.
- Count by complaint, not by number of times words appear.
- A complaint may belong to multiple reasons if it contains multiple distinct issues.
- Merge duplicate or near-duplicate reasons.
- Use concise customer-facing reason names.
- Sort by count descending.
- Return VALID JSON only.
- Use double quotes only.
- Do not use markdown.
- Do not explain.

Return format:
{{
  "reason_counts": [
    {{
      "reason": "one sentence reason",
      "count": 0
    }}
  ]
}}

Complaint summary notes:
{summary_notes_concat}
"""
    return prompt


def run_batches(df, apigee_access_token, model, sleep_seconds=2):
    batch_results = []

    for start in range(0, len(df), BATCH_SIZE):
        end = min(start + BATCH_SIZE, len(df))
        batch_df = df.iloc[start:end].copy()

        print(f"Running batch rows {start} to {end - 1}")

        prompt = build_batch_prompt(batch_df)

        response_text = generate_response_gemini(
            apigee_access_token,
            model,
            prompt
        )

        try:
            response_json = clean_llm_json(response_text)
            reason_counts = response_json["reason_counts"]

            batch_results.append({
                "batch_start": start,
                "batch_end": end - 1,
                "reason_counts": reason_counts,
                "raw_response": response_text
            })

        except Exception as e:
            print(f"Failed to parse batch {start}-{end - 1}")
            print(e)
            print(response_text[:2000])

            batch_results.append({
                "batch_start": start,
                "batch_end": end - 1,
                "reason_counts": None,
                "raw_response": response_text,
                "error": str(e)
            })

        time.sleep(sleep_seconds)

    return batch_results
