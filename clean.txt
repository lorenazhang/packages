WITH comp AS (
    SELECT
        LOWER(TRIM(conn_id)) AS conn_id,

        case_nr,
        complaint_id,
        complaint_type_desc,
        wf_recd_dt,
        created_dt,
        prim_complaint_category,
        prim_complaint_reason,
        product_category,
        product_type_desc,
        submitter_lob,
        submitter_sub_lob,

        1 AS comp_flag,

        CASE
            WHEN complaint_type_desc = 'Service Complaint' THEN 1
            ELSE 0
        END AS service_flag,

        CASE
            WHEN complaint_type_desc = 'Escalated Complaint' THEN 1
            ELSE 0
        END AS escalated_flag

    FROM EIW_I_CEDA_S1.COMPLAINTS_ECMP_REGISTRY_DLY_CS
    WHERE conn_id IS NOT NULL
      AND (
            (wf_recd_dt >= DATE '2025-03-15'
             AND wf_recd_dt < DATE '2025-06-01')
         OR (wf_recd_dt >= DATE '2026-03-15'
             AND wf_recd_dt < DATE '2026-06-01')
          )
),

call_data AS (
    SELECT
        LOWER(TRIM(connid)) AS connid,
        interactionid,
        primary_lob,
        sublob,
        calldriver1,
        interactiondatetime,

        TO_CHAR(CAST(interactiondatetime AS DATE), 'YYYYMM') AS month,

        CAST(interactiondatetime AS DATE)
            - ((CAST(interactiondatetime AS DATE) - DATE '1900-01-01') MOD 7) AS week,

        TO_CHAR(CAST(interactiondatetime AS DATE), 'YYYYMMDD') AS day

    FROM EIW_U_CEDA_VOCAL_V1.AMP_CALLDRIVERS2_0
    WHERE model_lang = 'English'
      AND calldriver1 NOT IN ('No Conversation', 'No Reason Disclosed')
      AND connid IS NOT NULL
      AND (
            (interactiondatetime >= TIMESTAMP '2025-03-15 00:00:00'
             AND interactiondatetime < TIMESTAMP '2025-06-01 00:00:00')
         OR (interactiondatetime >= TIMESTAMP '2026-03-15 00:00:00'
             AND interactiondatetime < TIMESTAMP '2026-06-01 00:00:00')
          )
      AND calldriver1 IN ({target_driver_str})
),

merged AS (
    SELECT
        c.connid,
        c.interactionid,
        c.primary_lob,
        c.sublob,
        c.calldriver1,
        c.interactiondatetime,
        c.month,
        c.week,
        c.day,

        comp.case_nr,
        comp.complaint_id,
        comp.complaint_type_desc,
        comp.wf_recd_dt,
        comp.created_dt,
        comp.prim_complaint_category,
        comp.prim_complaint_reason,
        comp.product_category,
        comp.product_type_desc,
        comp.submitter_lob,
        comp.submitter_sub_lob,

        COALESCE(comp.comp_flag, 0) AS comp_flag,
        COALESCE(comp.service_flag, 0) AS service_flag,
        COALESCE(comp.escalated_flag, 0) AS escalated_flag,

        CAST(comp.created_dt AS DATE) - CAST(comp.wf_recd_dt AS DATE) AS gap_days

    FROM call_data c
    LEFT JOIN comp
      ON c.connid = comp.conn_id
)

SELECT
    week,
    calldriver1,
    sublob,

    COUNT(*) AS call_count,

    SUM(comp_flag) AS complaint_count,
    SUM(service_flag) AS service_complaint_count,
    SUM(escalated_flag) AS escalated_complaint_count,

    AVG(comp_flag) AS complaint_conversion_rate,
    AVG(service_flag) AS service_conversion_rate,
    AVG(escalated_flag) AS escalated_conversion_rate,

    MEDIAN(gap_days) AS median_gap_days

FROM merged
GROUP BY
    week,
    calldriver1,
    sublob
ORDER BY
    calldriver1,
    sublob,
    week;
