
model = "gemini-2.5-flash"

batch_results = run_batches(
    df=df,
    apigee_access_token=apigee_access_token,
    model=model
)

rows = []

for result in batch_results:
    if result["reason_counts"] is None:
        continue

    for item in result["reason_counts"]:
        rows.append({
            "batch_start": result["batch_start"],
            "batch_end": result["batch_end"],
            "reason": item.get("reason"),
            "count": item.get("count")
        })

df_batch_reasons = pd.DataFrame(rows)

df_batch_reasons.head()


def build_combine_prompt(df_batch_reasons):
    batch_json = df_batch_reasons[["reason", "count"]].to_dict(orient="records")

    prompt = f"""
Task:
You are given reason counts extracted from multiple batches of complaint summary notes.

Combine them into one final consolidated reason count table.

Instructions:
- Merge duplicate, overlapping, or near-duplicate reasons.
- Sum the counts for merged reasons.
- Use concise customer-facing reason names.
- Sort by count descending.
- Return VALID JSON only.
- Use double quotes only.
- Do not use markdown.
- Do not explain.

Return format:
{{
  "reason_counts": [
    {{
      "reason": "one sentence reason",
      "count": 0
    }}
  ]
}}

Batch reason counts:
{json.dumps(batch_json, ensure_ascii=False)}
"""
    return prompt

combine_prompt = build_combine_prompt(df_batch_reasons)

final_response_text = generate_response_gemini(
    apigee_access_token,
    model,
    combine_prompt
)

final_json = clean_llm_json(final_response_text)

df_final_reasons = pd.DataFrame(final_json["reason_counts"])

df_final_reasons = df_final_reasons.sort_values(
    "count",
    ascending=False
).reset_index(drop=True)
